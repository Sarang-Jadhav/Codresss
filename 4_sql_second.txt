CREATE TABLE areas(
   radius NUMBER,
   area   NUMBER
);
/


DECLARE
    r NUMBER;
    a NUMBER;
BEGIN
    FOR r IN 5..9 LOOP
        a := 3.14 * r * r;
        
        INSERT INTO areas(radius, area)
        VALUES(r, a);

        DBMS_OUTPUT.PUT_LINE('Radius = '||r||'  Area = '||a);
    END LOOP;
END;
/
